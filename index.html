Temperature Conversion

This ruby program will prompt the user for a temperature in degrees Celsius and let the user know what the corresponding temperature is in Fahrenheit. 

				
					def c_to_f(temp)
					return (temp * 1.8) + 32
					end

					print "Enter temperature in degrees Celsius: "
					celsius = gets.to_f
					fahrenheit = c_to_f(celsius)
					puts "#{celsius} degrees C is equal to #{fahrenheit} degrees F"
				
			

				
					# Ordinal challenge, turning "18" into "18th" or "21" into "21st"

					def units_column_of(integer)
						return integer % 10
					end

					def tens_column_of(integer)
					# turn integer into a float & div by 10 to shift tens col to units col
						float = integer.to_f / 10
					#	turn float back into an integer & modulo 10 to return units col
						return float.to_i % 10
					end

					def correct_suffix(integer)
						if units_column_of(integer) == 1 && tens_column_of(integer) != 1
							return "st"
						elsif units_column_of(integer) == 2 && tens_column_of(integer) != 1
							return "nd"
						elsif units_column_of(integer) == 3 && tens_column_of(integer) != 1
							return "rd"
						else
							return "th"
						end
					end

					print "Enter a number: "
					number = gets.to_i
					puts "The ordinal of your number is #{number}" + correct_suffix(number)

Foobar

This ruby program is an adaptation of a classic technical interview problem which displays sequences of the Foobar pattern. 

				
					def run_foobar(total_number)
						loop_number = 1
						while loop_number <= total_number do
							if loop_number % 3 == 0 && loop_number % 5 == 0
								puts "Foobar"
							elsif loop_number % 3 == 0
								puts "Foo"
							elsif loop_number % 5 == 0
								puts "Bar"
							else
								puts loop_number.to_s
							end
							loop_number = loop_number + 1
						end
					end

					print "How many items do you want to see? "
					items = gets.to_i
					run_foobar(items)
